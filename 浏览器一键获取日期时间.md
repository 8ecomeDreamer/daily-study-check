let d = new Date();
let date = d.getFullYear() + "-" + String(d.getMonth()+1).padStart(2,0) + "-" + String(d.getDate()).padStart(2,0);
let time = String(d.getHours()).padStart(2,0) + ":" + String(d.getMinutes()).padStart(2,0);
let endYear = new Date(d.getFullYear(), 11, 31);
let leftDays = Math.ceil((endYear - d) / (1000*60*60*24));

console.log(`# 📅 每日打卡 | ${date}
> 节奏：工作日1h加练 | 周日3h复盘
⏳ 距离今年结束：<span style="font-size:2em; font-weight:bold; color:red;"> ${leftDays} </span> 天

⏰ 学习开始：${time}
⏰ 学习结束：${time}`);
